local function GetEFRating(winds)
    if winds < 35 then return "EF1"
    elseif winds < 80  then return "EF2"
    elseif winds < 100 then return "EF3"
    elseif winds < 120 then return "EF4"
    else return "EF5" end
end

local EF_STOPS = {
    {w=0,   c=Color3.fromRGB(0,220,0)},
    {w=35,  c=Color3.fromRGB(180,255,0)},
    {w=80,  c=Color3.fromRGB(255,220,0)},
    {w=100, c=Color3.fromRGB(255,130,0)},
    {w=120, c=Color3.fromRGB(255,40,0)},
    {w=140, c=Color3.fromRGB(180,0,0)},
}
local function GetEFColor(winds)
    winds = math.max(0, winds)
    if winds <= EF_STOPS[1].w then return EF_STOPS[1].c end
    if winds >= EF_STOPS[#EF_STOPS].w then return EF_STOPS[#EF_STOPS].c end
    for i = 1, #EF_STOPS - 1 do
        local a, b = EF_STOPS[i], EF_STOPS[i+1]
        if winds >= a.w and winds <= b.w then
            return a.c:Lerp(b.c, (winds - a.w) / (b.w - a.w))
        end
    end
    return EF_STOPS[#EF_STOPS].c
end

local function SaveAtmosphere()
    local a = Lighting:FindFirstChildOfClass("Atmosphere"); if not a then return end
    SavedAtmosphere = { Density=a.Density, Offset=a.Offset, Haze=a.Haze, Glare=a.Glare }
end
local function ApplyAtmosphereFullBright()
    local a = Lighting:FindFirstChildOfClass("Atmosphere"); if not a then return end
    a.Density=0; a.Offset=0; a.Haze=2; a.Glare=2
end
local function RestoreAtmosphere()
    local a = Lighting:FindFirstChildOfClass("Atmosphere")
    if not a or not next(SavedAtmosphere) then return end
    a.Density=SavedAtmosphere.Density; a.Offset=SavedAtmosphere.Offset
    a.Haze=SavedAtmosphere.Haze; a.Glare=SavedAtmosphere.Glare
end

    local sc = sr:FindFirstChild("clouds"); if sc then sc:ClearAllChildren() end
    local sf = sr:FindFirstChild("storms")

-- ==========================================
-- Исправленный GetGroundY для stealth/обычного
-- + Исключаем все части торнадо и clouds
-- + Исключаем Corn/Field1/Wheat в map_related.landscape
-- + Исключаем маркеры trail/predict
-- ==========================================
local raycastParams = RaycastParams.new()
raycastParams.FilterType = Enum.RaycastFilterType.Exclude
raycastParams.IgnoreWater = false

local RAYCAST_ORIGIN_Y = 10000
local RAYCAST_LENGTH   = 22000

local function GetGroundY(pos)
    local char    = LocalPlayer.Character
    local exclude = {}

    if char then table.insert(exclude, char) end
    if StealthVisualRig and StealthVisualRig.Parent then
        table.insert(exclude, StealthVisualRig)
    end
    if FlightAnchor and FlightAnchor.Parent then
        table.insert(exclude, FlightAnchor)
    end

    -- Исключаем всё, что связано с визуалом торнадо
    local sr = workspace:FindFirstChild("storm_related")
    if sr then
        local stormsFolder = sr:FindFirstChild("storms")
        local cloudsFolder = sr:FindFirstChild("clouds")
        if stormsFolder then table.insert(exclude, stormsFolder) end
        if cloudsFolder then table.insert(exclude, cloudsFolder) end

        -- Дополнительно все "cloud/curtain"-узлы в storm_related
        for _, child in ipairs(sr:GetChildren()) do
            local n = string.lower(child.Name)
            if n:find("cloud") or n:find("curtain") then
                table.insert(exclude, child)
            end
        end
    end

    -- Игнорируем объекты внутри workspace.map_related.landscape:
    -- Corn, Field1, Wheat (по имени)
    local mapRelated = workspace:FindFirstChild("map_related")
    local landscape = mapRelated and mapRelated:FindFirstChild("landscape")
    if landscape then
        for _, obj in ipairs(landscape:GetDescendants()) do
            if obj.Name == "Corn" or obj.Name == "Field1" or obj.Name == "Wheat" then
                table.insert(exclude, obj)
            end
        end
    end

    -- Игнорируем все парты/модели маркеров пути и предикта
    if MarkerFolder and MarkerFolder.Parent then
        table.insert(exclude, MarkerFolder)
    end

    local pe = Camera:FindFirstChild("ParticleEffects")
    if pe then
        for _, child in pairs(pe:GetChildren()) do
            table.insert(exclude, child)
        end
    end

    raycastParams.FilterDescendantsInstances = exclude

    local origin = Vector3.new(pos.X, RAYCAST_ORIGIN_Y, pos.Z)
    local result = workspace:Raycast(origin, Vector3.new(0, -RAYCAST_LENGTH, 0), raycastParams)
    if result then return result.Position.Y end
    return -math.huge
end

local function DestroyStealthVisualRig()
    if StealthVisualRig and StealthVisualRig.Parent then
        StealthVisualRig:Destroy()
    end
    StealthVisualRig = nil
end

local function CreateStealthVisualRig()
    DestroyStealthVisualRig()
    local char = LocalPlayer.Character; if not char then return end
    local oldArch = char.Archivable
    char.Archivable = true
    local ok, clone = pcall(function() return char:Clone() end)
    char.Archivable = oldArch
    if not ok or not clone then return end
    clone.Name = "StealthVisualRig"
    for _, d in ipairs(clone:GetDescendants()) do
        if d:IsA("Script") or d:IsA("LocalScript") then d:Destroy()
        elseif d:IsA("BasePart") then
            d.Anchored=true; d.CanCollide=false; d.CanTouch=false; d.Massless=true
        end
    end
    clone.Parent = workspace
    StealthVisualRig = clone
end

local function UpdateStealthVisualRig(anchorCF, yaw)
    if not StealthVisualRig or not StealthVisualRig.Parent then return end
    local targetCF = CFrame.new(anchorCF.Position) * CFrame.fromEulerAnglesYXZ(0, yaw, 0)
    pcall(function() StealthVisualRig:PivotTo(targetCF) end)
end

-- Полёт: включение / выключение
-- ==========================================
local function EnableFlight()
    if State.FlightEnabled then return end
    local char = LocalPlayer.Character
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    local hum  = char and char:FindFirstChild("Humanoid")
    if not hrp or not hum then return end
    if hum.Sit or hum.SeatPart then return end

    hum.PlatformStand = false
    hrp.AssemblyLinearVelocity  = Vector3.zero
    hrp.AssemblyAngularVelocity = Vector3.zero

    FlightAnchor = Instance.new("Part")
    FlightAnchor.Name="FlightAnchor"; FlightAnchor.Size=Vector3.new(1,1,1)
    FlightAnchor.Anchored=true; FlightAnchor.CanCollide=false
    FlightAnchor.CanTouch=false; FlightAnchor.CastShadow=false
    FlightAnchor.Transparency=1; FlightAnchor.CFrame=hrp.CFrame
    FlightAnchor.Parent=workspace

    if State.StealthFlight then
        StealthOriginCF = hrp.CFrame
        local _, cy, _ = Camera.CFrame:ToEulerAnglesYXZ()
        stealthCamYaw=cy; stealthCamPitch=0.25
        stealthCamDist=CAM_DIST_DEFAULT; stealthRotating=false
        CreateStealthVisualRig()
        SetStealthCamera()
    else
        hum.PlatformStand = true
        StealthOriginCF = nil
        DestroyStealthVisualRig()
    end

    State.FlightEnabled = true
    Rayfield:Notify({ Title="Полёт", Content="Полёт включён. WASD + Space/LCtrl", Duration=3, Image=4483362458 })
end

local function DisableFlight()
    if not State.FlightEnabled then return end

    if State.YLock and State.LockedCFrame then
        Rayfield:Notify({ Title="Полёт", Content="Сначала отключи фиксацию осей!", Duration=3.5, Image=4483362458 })
        return
    end

    local char = LocalPlayer.Character
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    local hum  = char and char:FindFirstChild("Humanoid")

    if State.StealthFlight then
        if hrp and FlightAnchor then
            hrp.CFrame = CFrame.new(FlightAnchor.CFrame.Position)
                * CFrame.fromEulerAnglesYXZ(0, stealthCamYaw or 0, 0)
            hrp.AssemblyLinearVelocity  = Vector3.zero
            hrp.AssemblyAngularVelocity = Vector3.zero
        end
        StealthOriginCF = nil
        DestroyStealthVisualRig()
    end

    if hum then hum.PlatformStand = false end
    State.FlightEnabled = false
    flightMoveDir = Vector3.zero
    stealthRotating = false

    if FlightAnchor then FlightAnchor:Destroy(); FlightAnchor=nil end
    RestorePlayerCamera()
    Rayfield:Notify({ Title="Полёт", Content="Полёт выключен.", Duration=2.5, Image=4483362458 })
end

UIS.InputBegan:Connect(function(input, gp)
    if gp then return end

    if State.FlightEnabled and State.StealthFlight
       and input.UserInputType == Enum.UserInputType.MouseButton2 then
        stealthRotating = true
        UIS.MouseBehavior    = Enum.MouseBehavior.LockCurrentPosition
        UIS.MouseIconEnabled = false
        return
    end

    if keyDebounce[input.KeyCode] then return end
    keyDebounce[input.KeyCode] = true
    task.delay(0.25, function() keyDebounce[input.KeyCode] = nil end)

    local kc = input.KeyCode

    local fkStr = FlightKeybind.CurrentKeybind
    if fkStr and fkStr ~= "" then
        local ok, fk = pcall(function() return Enum.KeyCode[fkStr] end)
        if ok and kc == fk then
            if State.FlightEnabled then DisableFlight() else EnableFlight() end
        end
    end

    local lkStr = LockKeybind.CurrentKeybind
    if lkStr and lkStr ~= "" then
        local ok, lk = pcall(function() return Enum.KeyCode[lkStr] end)
        if ok and kc == lk then
            State.YLock = not State.YLock
            if State.YLock then
                local root = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if root then
                    State.LockedCFrame = root.CFrame
                    Rayfield:Notify({ Title="Фиксация", Content="Оси зафиксированы!", Duration=2.5, Image=4483362458 })
                end
            else
                State.LockedCFrame = nil
                Rayfield:Notify({ Title="Фиксация", Content="Фиксация снята.", Duration=2, Image=4483362458 })
            end
        end
    end
end)

UIS.InputEnded:Connect(function(input, gp)
    if gp then return end
    if input.UserInputType == Enum.UserInputType.MouseButton2 then
        stealthRotating = false
        if State.FlightEnabled and State.StealthFlight then
            UIS.MouseBehavior    = Enum.MouseBehavior.Default
            UIS.MouseIconEnabled = true
        end
    end
end)

UIS.InputChanged:Connect(function(input)
    if not (State.FlightEnabled and State.StealthFlight) then return end
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        if stealthRotating then
            stealthCamYaw   = stealthCamYaw   - input.Delta.X * 0.003
            stealthCamPitch = math.clamp(stealthCamPitch - input.Delta.Y * 0.003, -1.2, 1.2)
        end
    elseif input.UserInputType == Enum.UserInputType.MouseWheel then
        stealthCamDist = math.clamp(
            stealthCamDist - input.Position.Z * CAM_ZOOM_STEP,
            CAM_DIST_MIN, CAM_DIST_MAX
        )
    end
end)

local Drawings = {}

local function CreateESP(instance, name, espType)
    if Drawings[instance] then return end
    local box = Drawing.new("Square")
    box.Visible=false; box.Thickness=(espType=="player") and 1.5 or 2; box.Filled=false
    local text = Drawing.new("Text")
    text.Visible=false; text.Center=true; text.Outline=true
    text.Size=(espType=="player") and 14 or 15
    local efText = nil
    if espType == "tornado" then
        efText = Drawing.new("Text")
        efText.Visible=false; efText.Center=true; efText.Outline=true
        efText.Size=13; efText.Color=Color3.fromRGB(255,255,255)
    end
    Drawings[instance] = { Box=box, Text=text, EFText=efText, Type=espType, Name=name }
end

local function UpdateESP()
    for instance, esp in pairs(Drawings) do
        if not instance or not instance.Parent then
            esp.Box:Remove(); esp.Text:Remove()
            if esp.EFText then esp.EFText:Remove() end
            Drawings[instance]=nil; continue
        end
        local espType = esp.Type
        local enabled =
            (espType=="player"  and State.ESP_Players) or
            (espType=="tornado" and State.ESP_Tornado)  or
            (espType=="probe"   and State.ESP_Probes)

        local anchor
        if espType=="player" then
            anchor = instance:FindFirstChild("HumanoidRootPart")
        elseif espType=="tornado" then
            anchor = GetScanPart(instance)
        else
            anchor = instance.PrimaryPart or instance:FindFirstChildWhichIsA("BasePart")
        end

        local color
        if espType=="player" then
            color = State.ESP_Color_Players
        elseif espType=="tornado" then
            local winds = GetWindsValue(instance)
            color = (State.ESP_EFColor and winds) and GetEFColor(winds) or State.ESP_Color_Tornado
        else
            color = State.ESP_Color_Probes
        end

        esp.Box.Color=color; esp.Text.Color=color
        if enabled and anchor then
            local vec, onScreen = Camera:WorldToViewportPoint(anchor.Position)
            if onScreen then
                local halfH = (espType=="player") and 2.8 or 3
                local tv = Camera:WorldToViewportPoint(anchor.Position+Vector3.new(0, halfH,0))
                local bv = Camera:WorldToViewportPoint(anchor.Position+Vector3.new(0,-halfH,0))
                local sH = math.clamp(math.abs(tv.Y-bv.Y),(espType=="player") and 14 or 10,(espType=="player") and 220 or 300)
                local sW = sH*((espType=="player") and 0.42 or 0.75)
                local tY = math.min(tv.Y,bv.Y)
                esp.Box.Size=Vector2.new(sW,sH); esp.Box.Position=Vector2.new(vec.X-sW/2,tY); esp.Box.Visible=true
                local dist=math.floor((Camera.CFrame.Position-anchor.Position).Magnitude)
                esp.Text.Text=State.ESP_Distance and string.format("%s [%dm]",esp.Name,dist) or esp.Name
                esp.Text.Position=Vector2.new(vec.X,tY-20); esp.Text.Visible=true
                if esp.EFText then
                    if State.ESP_ShowEF or State.ESP_ShowWinds then
                        local winds=GetWindsValue(instance)
                        local parts={}
                        if winds then
                            if State.ESP_ShowEF    then table.insert(parts,GetEFRating(winds)) end
                            if State.ESP_ShowWinds then table.insert(parts,string.format("%.0f m/s",winds)) end
                            esp.EFText.Color=State.ESP_EFColor and GetEFColor(winds) or Color3.fromRGB(255,255,255)
                        else
                            if State.ESP_ShowEF    then table.insert(parts,"EF?") end
                            if State.ESP_ShowWinds then table.insert(parts,"? m/s") end
                            esp.EFText.Color=Color3.fromRGB(200,200,200)
                        end
                        esp.EFText.Text=table.concat(parts,"  |  ")
                        esp.EFText.Position=Vector2.new(vec.X,tY+sH+4); esp.EFText.Visible=true
                    else
                        esp.EFText.Visible=false
                    end
                end
            else
                esp.Box.Visible=false; esp.Text.Visible=false
                if esp.EFText then esp.EFText.Visible=false end
            end
        else
            esp.Box.Visible=false; esp.Text.Visible=false
            if esp.EFText then esp.EFText.Visible=false end
        end
    end
end

local function MakePart(pos, hlColor, lifetime)
    local part = Instance.new("Part")
    part.Size=Vector3.new(5,5,5); part.Position=pos; part.Anchored=true
    part.CanCollide=false; part.CastShadow=false; part.Transparency=0
    part.Color=Color3.new(0,0,0); part.Material=Enum.Material.SmoothPlastic
    part.Parent=EnsureMarkerFolder()

    local hl=Instance.new("Highlight")
    hl.FillColor=hlColor; hl.OutlineColor=hlColor
    hl.FillTransparency=0; hl.OutlineTransparency=0
    hl.DepthMode=Enum.HighlightDepthMode.AlwaysOnTop; hl.Adornee=part; hl.Parent=part
    if lifetime then Debris:AddItem(part,lifetime) end
    return part
end

local SMOOTH_FACTOR=0.08
local PREDICT_STEPS=5
local PREDICT_FRAMES=120
local smoothVelocity={}

local function EnsurePredictParts(storm)
    if not PredictParts[storm] then PredictParts[storm]={} end
    for i=1,PREDICT_STEPS do
        if not PredictParts[storm][i] or not PredictParts[storm][i].Parent then
            PredictParts[storm][i]=MakePart(Vector3.new(0,-9999,0),State.PredictColor,nil)
        end
        local hl=PredictParts[storm][i]:FindFirstChildOfClass("Highlight")
        if hl then hl.FillColor=State.PredictColor; hl.OutlineColor=State.PredictColor end
    end
end
local function UpdatePredictParts(storm,curPos,rawVelocity)
    if not smoothVelocity[storm] then smoothVelocity[storm]=rawVelocity end
    local sv=smoothVelocity[storm]
    smoothVelocity[storm]=Vector3.new(
        sv.X+(rawVelocity.X-sv.X)*SMOOTH_FACTOR,
        sv.Y+(rawVelocity.Y-sv.Y)*SMOOTH_FACTOR,
        sv.Z+(rawVelocity.Z-sv.Z)*SMOOTH_FACTOR
    )
    if smoothVelocity[storm].Magnitude<0.005 then return end
    EnsurePredictParts(storm)
    for i=1,PREDICT_STEPS do
        PredictParts[storm][i].Position=curPos+smoothVelocity[storm]*(PREDICT_FRAMES*(i/PREDICT_STEPS))
    end
end
local function DestroyPredictParts(storm)
    if not PredictParts[storm] then return end
    for _,p in ipairs(PredictParts[storm]) do pcall(function()p:Destroy()end) end
    PredictParts[storm]=nil; smoothVelocity[storm]=nil
end

local trailPos={}
local prevPos={}

RunService.RenderStepped:Connect(function(dt)
    if State.FullBright then
        Lighting.Ambient=Color3.new(1,1,1); Lighting.ColorShift_Bottom=Color3.new(1,1,1)
        Lighting.ColorShift_Top=Color3.new(1,1,1); Lighting.GlobalShadows=false; Lighting.ClockTime=14
    end

    if State.SpeedEnabled and not State.FlightEnabled then
        local hum=LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Humanoid")
        if hum then hum.WalkSpeed=29 end
    end

    if State.FlightEnabled and not FlightAnchor then DisableFlight() end

    if State.FlightEnabled and FlightAnchor then
        local lookVec, rightVec
        if State.StealthFlight then
            local cf=CFrame.fromEulerAnglesYXZ(stealthCamPitch,stealthCamYaw,0)
            lookVec=cf.LookVector; rightVec=cf.RightVector
        else
            lookVec=Camera.CFrame.LookVector; rightVec=Camera.CFrame.RightVector
        end

        local fwd=Vector3.new(lookVec.X,0,lookVec.Z)
        local rgt=Vector3.new(rightVec.X,0,rightVec.Z)
        if fwd.Magnitude>0 then fwd=fwd.Unit end
        if rgt.Magnitude>0 then rgt=rgt.Unit end

        local dir=Vector3.zero
        if UIS:IsKeyDown(Enum.KeyCode.W)           then dir=dir+fwd end
        if UIS:IsKeyDown(Enum.KeyCode.S)           then dir=dir-fwd end
        if UIS:IsKeyDown(Enum.KeyCode.A)           then dir=dir-rgt end
        if UIS:IsKeyDown(Enum.KeyCode.D)           then dir=dir+rgt end
        if UIS:IsKeyDown(Enum.KeyCode.Space)       then dir=dir+Vector3.new(0,1,0) end
        if UIS:IsKeyDown(Enum.KeyCode.LeftControl) then dir=dir-Vector3.new(0,1,0) end
        if dir.Magnitude>0 then dir=dir.Unit end

        flightMoveDir=dir*State.FlightSpeed*dt
        local newPos=(FlightAnchor.CFrame+flightMoveDir).Position

        local groundY=GetGroundY(newPos)
        if groundY ~= -math.huge then
            local minY=groundY+FLIGHT_MIN_HEIGHT
            if newPos.Y < minY then
                newPos=Vector3.new(newPos.X,minY,newPos.Z)
            end
        end

        FlightAnchor.CFrame=CFrame.new(newPos)

        if State.StealthFlight then
            if Camera.CameraType ~= Enum.CameraType.Scriptable then SetStealthCamera() end
            local anchorPos=FlightAnchor.CFrame.Position
            local back=CFrame.fromEulerAnglesYXZ(0,stealthCamYaw,0).LookVector*-1
            local camPos=anchorPos+back*stealthCamDist+Vector3.new(0,CAM_HEIGHT,0)
            Camera.CFrame=CFrame.new(camPos,anchorPos+Vector3.new(0,1.5,0))
            UpdateStealthVisualRig(FlightAnchor.CFrame,stealthCamYaw)
        end
    end

    for _,player in pairs(Players:GetPlayers()) do
        if player~=LocalPlayer and player.Character then
            CreateESP(player.Character,player.Name,"player")
        end
    end

    local sf=GetStormsFolder()
    if sf then
        for _,storm in pairs(sf:GetChildren()) do
            CreateESP(storm,storm.Name,"tornado")
            local tp=GetTrackPart(storm)
            if not tp then continue end
            local curPos=tp.Position
            if not prevPos[storm] then prevPos[storm]=curPos; trailPos[storm]=curPos end
            local rawVel=curPos-prevPos[storm]
            prevPos[storm]=curPos
            if State.TornadoTrail and (curPos-trailPos[storm]).Magnitude>=2 then
                MakePart(curPos,State.TrailColor,20); trailPos[storm]=curPos
            end
            if State.TornadoPredict then UpdatePredictParts(storm,curPos,rawVel)
            else DestroyPredictParts(storm) end
        end
    end

    local pf=GetProbesFolder()
    if pf then
        for _,probe in pairs(pf:GetChildren()) do
            if probe:IsA("Model") then CreateESP(probe,probe.Name,"probe") end
        end
    end

    for storm in pairs(PredictParts) do
        if not storm or not storm.Parent then DestroyPredictParts(storm) end
    end

    UpdateESP()
    UpdateLockIndicator()
end)

RunService.Heartbeat:Connect(function()
    local char=LocalPlayer.Character
    local hrp=char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    if State.FlightEnabled and FlightAnchor then
        local hum=char:FindFirstChild("Humanoid")
        if hum and (hum.Sit or hum.SeatPart) then DisableFlight(); return end

        if State.StealthFlight then
            if hum then hum.PlatformStand=false end
            if StealthOriginCF then hrp.CFrame=StealthOriginCF end
            hrp.AssemblyLinearVelocity=Vector3.zero
            hrp.AssemblyAngularVelocity=Vector3.zero
        else
            if hum and not hum.PlatformStand then hum.PlatformStand=true end
            local _,camYaw,_=Camera.CFrame:ToEulerAnglesYXZ()
            hrp.CFrame=CFrame.new(FlightAnchor.CFrame.Position)*CFrame.fromEulerAnglesYXZ(0,camYaw,0)
            hrp.AssemblyLinearVelocity=Vector3.zero
            hrp.AssemblyAngularVelocity=Vector3.zero
        end

    elseif State.YLock and State.LockedCFrame then
        local lockY=HasAxis("Только Y")
        local lockXZ=HasAxis("X и Z")
        local lockAll=lockY and lockXZ
        local cur=hrp.Position
        local lk=State.LockedCFrame.Position
        if lockAll then
            hrp.CFrame=State.LockedCFrame
        elseif lockY or lockXZ then
            local newX=lockXZ and lk.X or cur.X
            local newY=lockY  and lk.Y or cur.Y
            local newZ=lockXZ and lk.Z or cur.Z
            if lockXZ then
                local _,ly,_=State.LockedCFrame:ToEulerAnglesYXZ()
                hrp.CFrame=CFrame.new(newX,newY,newZ)*CFrame.fromEulerAnglesYXZ(0,ly,0)
            else
                local rot=CFrame.fromMatrix(Vector3.zero,hrp.CFrame.RightVector,hrp.CFrame.UpVector)
                hrp.CFrame=CFrame.new(newX,newY,newZ)*rot
            end
        end
        hrp.AssemblyLinearVelocity=Vector3.zero
        hrp.AssemblyAngularVelocity=Vector3.zero
    end
end)
